\begin{defn}[Probability]{Probability}
	An exact mathematical \textbf{probability}\index{probability} $p$ can be defined by the asymptotic frequency of successes $N_s$ in the limit of an infinite number of trials $N$.
 \begin{align}
  p = \lim_{N\rightarrow\inf}\frac{N_s}{N}
 \end{align}
\end{defn}
